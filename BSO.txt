CREATE DATABASE BSO
GO

USE BSO
GO

--DROP TABLE DefaultMensaje 
CREATE TABLE DefaultMensaje(
	Servidor		VARCHAR(30),
	"Fecha"	VARCHAR(30),
	Nombre	VARCHAR(30),
	Cedula	VARCHAR(30)
)
GO

--DROP PROCEDURE insertarMsj
CREATE PROCEDURE insertarMsj(
	@Servidor VARCHAR(30),
	@Fecha VARCHAR(30),
	@Cedula VARCHAR(30),
	@Nombre VARCHAR(30))
AS 
begin
	INSERT INTO DefaultMensaje(Servidor,"Fecha",Cedula,Nombre)
	VALUES (@Servidor,@Fecha,@Nombre,@Cedula)
end
GO

select * from DefaultMensaje
delete DefaultMensaje  where Servidor = Servidor
